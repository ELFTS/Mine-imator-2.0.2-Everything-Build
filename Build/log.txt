uniform sampler2D u_screen;   // scene texture
uniform float2 u_resolution;    // screen resolution
uniform float u_power;        // edge sensitivity multiplier

void main() {
    float2 texel = 1.0 / u_resolution;

    // sample neighborhood
    float4 cNW = texture2D(u_screen, _vars.vTexCoord + float2_(-1.0, -1.0) * texel);
    float4 cNE = texture2D(u_screen, _vars.vTexCoord + float2_( 1.0, -1.0) * texel);
    float4 cSW = texture2D(u_screen, _vars.vTexCoord + float2_(-1.0,  1.0) * texel);
    float4 cSE = texture2D(u_screen, _vars.vTexCoord + float2_( 1.0,  1.0) * texel);
    float4 cM  = texture2D(u_screen, _vars.vTexCoord);

    // extract rgb
    float3 rgbNW = cNW.rgb;
    float3 rgbNE = cNE.rgb;
    float3 rgbSW = cSW.rgb;
    float3 rgbSE = cSE.rgb;
    float3 rgbM  = cM.rgb;

    // luma calculation (perceptual brightness)
    float lumaNW = dot(rgbNW, float3_(0.299, 0.587, 0.114));
    float lumaNE = dot(rgbNE, float3_(0.299, 0.587, 0.114));
    float lumaSW = dot(rgbSW, float3_(0.299, 0.587, 0.114));
    float lumaSE = dot(rgbSE, float3_(0.299, 0.587, 0.114));
    float lumaM  = dot(rgbM,  float3_(0.299, 0.587, 0.114));

    // find edge contrast
    float lumaMin = min(lumaM, min(min(lumaNW, lumaNE), min(lumaSW, lumaSE)));
    float lumaMax = max(lumaM, max(max(lumaNW, lumaNE), max(lumaSW, lumaSE)));

    // if contrast is low, skip AA (keeps sharpness)
    if (lumaMax - lumaMin < max(0.0312, lumaMax * 0.125 * u_power)) {
        _out.Color0 = cM; // keep original alpha
        return;
    }

    // edge direction
    float2 dir;
    dir.x = -((lumaNW + lumaNE) - (lumaSW + lumaSE));
    dir.y =  ((lumaNW + lumaSW) - (lumaNE + lumaSE));

    // reduce length (avoid oversmoothing)
    float dirReduce = max((lumaNW + lumaNE + lumaSW + lumaSE) * (0.25 * 1.0/8.0), 1.0/128.0);
    float rcpDirMin = 1.0 / (min(abs(dir.x), abs(dir.y)) + dirReduce);
    dir = clamp(dir * rcpDirMin, -8.0, 8.0) * texel;

    // blend samples along edge direction
    float4 cA = 0.5 * (
        texture2D(u_screen, _vars.vTexCoord + dir * (1.0/3.0 - 0.5)) +
        texture2D(u_screen, _vars.vTexCoord + dir * (2.0/3.0 - 0.5))
    );

    float4 cB = cA * 0.5 + 0.25 * (
        texture2D(u_screen, _vars.vTexCoord + dir * -0.5) +
        texture2D(u_screen, _vars.vTexCoord + dir * 0.5)
    );

    float lumaB = dot(cB.rgb, float3_(0.299, 0.587, 0.114));

    _out.Color0 = ((lumaB < lumaMin) || (lumaB > lumaMax)) ? cA : cB;
}

struct Vars

	float4 gl_Position : SV_POSITION;
	float2 vTexCoord : A;

uniform sampler2D u_screen;   // scene texture
uniform float2 u_resolution;    // screen resolution
uniform float u_power;        // edge sensitivity multiplier

void main() {
    float2 texel = 1.0 / u_resolution;

    // sample neighborhood
    float4 cNW = texture2D(u_screen, _vars.vTexCoord + float2_(-1.0, -1.0) * texel);
    float4 cNE = texture2D(u_screen, _vars.vTexCoord + float2_( 1.0, -1.0) * texel);
    float4 cSW = texture2D(u_screen, _vars.vTexCoord + float2_(-1.0,  1.0) * texel);
    float4 cSE = texture2D(u_screen, _vars.vTexCoord + float2_( 1.0,  1.0) * texel);
    float4 cM  = texture2D(u_screen, _vars.vTexCoord);

    // extract rgb
    float3 rgbNW = cNW.rgb;
    float3 rgbNE = cNE.rgb;
    float3 rgbSW = cSW.rgb;
    float3 rgbSE = cSE.rgb;
    float3 rgbM  = cM.rgb;

    // luma calculation (perceptual brightness)
    float lumaNW = dot(rgbNW, float3_(0.299, 0.587, 0.114));
    float lumaNE = dot(rgbNE, float3_(0.299, 0.587, 0.114));
    float lumaSW = dot(rgbSW, float3_(0.299, 0.587, 0.114));
    float lumaSE = dot(rgbSE, float3_(0.299, 0.587, 0.114));
    float lumaM  = dot(rgbM,  float3_(0.299, 0.587, 0.114));

    // find edge contrast
    float lumaMin = min(lumaM, min(min(lumaNW, lumaNE), min(lumaSW, lumaSE)));
    float lumaMax = max(lumaM, max(max(lumaNW, lumaNE), max(lumaSW, lumaSE)));

    // if contrast is low, skip AA (keeps sharpness)
    if (lumaMax - lumaMin < max(0.0312, lumaMax * 0.125 * u_power)) {
        _out.Color0 = cM; // keep original alpha
        return;
    }

    // edge direction
    float2 dir;
    dir.x = -((lumaNW + lumaNE) - (lumaSW + lumaSE));
    dir.y =  ((lumaNW + lumaSW) - (lumaNE + lumaSE));

    // reduce length (avoid oversmoothing)
    float dirReduce = max((lumaNW + lumaNE + lumaSW + lumaSE) * (0.25 * 1.0/8.0), 1.0/128.0);
    float rcpDirMin = 1.0 / (min(abs(dir.x), abs(dir.y)) + dirReduce);
    dir = clamp(dir * rcpDirMin, -8.0, 8.0) * texel;

    // blend samples along edge direction
    float4 cA = 0.5 * (
        texture2D(u_screen, _vars.vTexCoord + dir * (1.0/3.0 - 0.5)) +
        texture2D(u_screen, _vars.vTexCoord + dir * (2.0/3.0 - 0.5))
    );

    float4 cB = cA * 0.5 + 0.25 * (
        texture2D(u_screen, _vars.vTexCoord + dir * -0.5) +
        texture2D(u_screen, _vars.vTexCoord + dir * 0.5)
    );

    float lumaB = dot(cB.rgb, float3_(0.299, 0.587, 0.114));

    _out.Color0 = ((lumaB < lumaMin) || (lumaB > lumaMax)) ? cA : cB;
}
